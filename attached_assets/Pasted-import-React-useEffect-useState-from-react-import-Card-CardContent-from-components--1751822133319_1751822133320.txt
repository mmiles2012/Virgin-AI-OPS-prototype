import React, { useEffect, useState } from "react";
import { Card, CardContent } from "@/components/ui/card";

const fetchLiveNetworkData = async () => {
  const response = await fetch("https://opensky-network.org/api/states/all");
  const json = await response.json();
  return json;
};

const fetchWeatherData = async () => {
  const response = await fetch("https://api.weatherapi.com/v1/current.json?key=YOUR_API_KEY&q=London");
  const json = await response.json();
  return json;
};

export default function AIOpsDashboard() {
  const [openskyData, setOpenskyData] = useState(null);
  const [weatherData, setWeatherData] = useState(null);

  useEffect(() => {
    fetchLiveNetworkData().then(setOpenskyData);
    fetchWeatherData().then(setWeatherData);
  }, []);

  const data = {
    onTimePerformance: "79%",
    cancellations: 12,
    diversions: 4,
    curfews: 6,
    alerts: [
      { id: 1, message: "Delay forecasted for AA123 due to weather" },
      { id: 2, message: "Diversion recommended for DL456" },
      { id: 3, message: "Tight connection detected for UA739" }
    ]
  };

  return (
    <div className="grid grid-cols-3 gap-4 p-4 bg-slate-900 text-white min-h-screen">
      <Card className="col-span-2">
        <CardContent>
          <h2 className="text-xl font-bold mb-2">Live Map View</h2>
          <div className="h-64 bg-slate-800 rounded-xl overflow-auto p-2 text-xs">
            {openskyData ? (
              <pre>{JSON.stringify(openskyData.states.slice(0, 5), null, 2)}</pre>
            ) : (
              <span className="text-slate-400">Loading live aircraft data...</span>
            )}
          </div>
        </CardContent>
      </Card>

      <Card>
        <CardContent>
          <h2 className="text-xl font-bold mb-2">Network Health</h2>
          <ul className="space-y-1">
            <li>On-Time Performance: {data.onTimePerformance}</li>
            <li>Cancellations: {data.cancellations}</li>
            <li>Diversions: {data.diversions}</li>
            <li>Curfews: {data.curfews}</li>
          </ul>
        </CardContent>
      </Card>

      <Card>
        <CardContent>
          <h2 className="text-xl font-bold mb-2">Digital Twin Alerts</h2>
          <ul className="space-y-2">
            {data.alerts.map(alert => (
              <li key={alert.id} className="bg-yellow-900 p-2 rounded-xl text-yellow-300">
                ⚠️ {alert.message}
              </li>
            ))}
          </ul>
        </CardContent>
      </Card>

      <Card>
        <CardContent>
          <h2 className="text-xl font-bold mb-2">Predictive Disruption Timeline</h2>
          <div className="h-32 bg-gradient-to-r from-green-600 via-yellow-500 to-red-500 rounded-xl flex items-center justify-center text-slate-900">
            [Disruption Timeline Chart Placeholder]
          </div>
        </CardContent>
      </Card>

      <Card>
        <CardContent>
          <h2 className="text-xl font-bold mb-2">Weather & Airspace</h2>
          <div className="h-32 bg-slate-800 rounded-xl overflow-auto p-2 text-xs">
            {weatherData ? (
              <pre>{JSON.stringify(weatherData.current, null, 2)}</pre>
            ) : (
              <span className="text-slate-400">Loading weather data...</span>
            )}
          </div>
        </CardContent>
      </Card>
    </div>
  );
}
